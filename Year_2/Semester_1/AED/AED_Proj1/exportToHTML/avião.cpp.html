#include "avião.h"
Aviao::Aviao(string m, string t, unsigned int c, unsigned int v) {
    matricula = std::move(m);
    tipo = std::move(t);
    capacidade = c;
    vagas = v;
}
string Aviao::getMatricula() {
    return matricula;
}
string Aviao::getTipo() {
    return tipo;
}
unsigned int Aviao::getCapacidade() const {
    return capacidade;
}
unsigned int Aviao::getVagas() const {
    return vagas;
}
