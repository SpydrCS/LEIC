#include "voo.h"
Voo::Voo(unsigned int n, string dp, unsigned int d, string o, string de) {
    numero = n;
    data_partida = std::move(dp);
    duracao = d;
    origem = std::move(o);
    destino = std::move(de);
}
unsigned int Voo::getNumero() const {
    return numero;
}
string Voo::getDataPartida() {
    return data_partida;
}
unsigned int Voo::getDuracao() const {
    return duracao;
}
string Voo::getOrigem() {
    return origem;
}
string Voo::getDestino() {
    return destino;
}
